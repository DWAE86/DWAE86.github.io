题目大意

nnn个aia_iai​，表示第i处有aia_iai​个块。每次可以选择下标的一个子集，使子集内的块减1，不能减块数为0的位置。要求选择n+1n+1n+1个子集至多，且使用子集两两不同。输出任意一个构造答案，减完所有aia_iai​。

样例输入输出

分析

一个2500的题在赛后我看的时候被我数十分钟一己之力干掉，一发入魂，还和题解方法不一样，还是挺吊的，虽然我都不太明白我的做法为什么对。。。

通过一番在草稿纸上发挥，我觉得应该对每个aia_iai​构造答案每一列。不知道怎么，我就得出按降序排列aia_iai​去求解。。。然后我就奇思妙想到阶梯式地减。。。

int main()
{

	for(int i = 1; i <= n; i++)
	{
		a[i] = {i, read()};
	}

	{

	}
	return 0;
}

D. Wrong Answer on test 233 (Easy/Hard Version) (1s 256Mb)

题目大意

nnn个选择题，每个都有kkk种选项。给出hih_ihi​为第i个题的答案，求有多少种答题方案，使得此方案向右移动一位（nnn->111）去答题比原情况得分严格高。mod998244353mod 998244353mod998244353。

n≤2⋅103(E)/2⋅105(H)n \leq 2\cdot 10^3(E)/2\cdot 10^5(H)n≤2⋅103(E)/2⋅105(H)

分析

Easy：n2n^2n2dp表示种数，一维前iii位，一维得分差，n⋅nn\cdot nn⋅n转移。

Hard：设移动后得分减移动前得分为ddd，每个位置选择答案为aia_iai​。对于每个位置hi=hi+1h_i=h_{i+1}hi​=hi+1​，如果，则无论如何设置，对ddd没有影响。否则可以使使得d加1，也可以使使得ddd减1，其他设置aia_iai​对ddd无影响。设可有影响的题数为ttt。枚举ttt里有iii题对答案无影响，则答案为如图。不能影响题随便选择，可有但无从k−2k-2k−2种里面选择，剩余组合数学。

强大的wzgy的代码教会我们，每一种d>0d>0d>0的情况，都一一对应一种d<0d<0d<0的情况，所以只需将所有情况计算出来，减去平局d=0d=0d=0，再除以222即可，wow！

代码

int main()
{
	n = read(), k = read();
	pre();
	for(int i = 1; i <= n; i++) a[i] = read();
	if(k == 1) //特判k=1时
	{
		printf("0\n");
		return 0;
	}
	for(int i = 1; i <= n; i++) if(a[i] != a[i%n+1]) ab++;//计算可改变d的题目数量
	int ans = 0;
	for(int i = 0; i*2 <= ab; i++) //求解平局情况
	{
		ans = (ans + 1LL*C(ab, i*2)*C(i*2, i)%mod*km(k-2, ab-i*2)%mod)%mod;
	}
	ans = 1LL*(km(k, ab)-ans+mod)*inv[2]%mod; //所以情况减去平局除以2
	ans = 1LL*ans*km(k, n-ab)%mod; //最后乘上不可改变d的题的每题k个随便选
	printf("%d\n", ans);
	return 0;
}

C. Arson In Berland Forest (2s 512Mb)

题目大意

n⋅mn\cdot mn⋅m 的'.' 和 'X' 图 ，'X'表示已被火烧的树，'.'则表示未被烧。燃烧情况是每棵燃烧的树通过1秒将八个方向的树烧着。构造一个可行答案，要求经过时间TTT最大，并输出第0时刻已经烧着的树是哪些，这些树的数量没有限制。

1≤n,m≤106,1≤n⋅m≤1061 \leq n,m \leq 10^6, 1 \leq n\cdot m \leq 10^61≤n,m≤106,1≤n⋅m≤106

样例输入输出

分析

显然二分时间TTT，关键在于check。从下到上，从右到左，求解右下方可燃烧范围。如果大于2∗T+12*T+12∗T+1，可以在右+T+T+T,下+T+T+T放初始燃烧树。最后判断给个每棵树是不是都被放置初始燃烧树燃烧。判断时，可以用一个巧方法，将燃烧等价于从左上角往右下燃烧2∗T+12*T+12∗T+1，所以判断时可以看左上三个方向那个剩余燃烧范围最大，也是前缀和。

代码

int ma[maxn], vis[maxn];
int check(int x)
{
	memset(vis, 0, sizeof(vis));
	int s = x<<1|1;
	for(int i = 1; i <= n; i++) //放置初始燃烧树
	for(int j = 1; j <= m; j++) if(ma[cor(i, j)] >= s) vis[cor(i, j)] = s;
	
	for(int i = 1; i <= n; i++) //将所有可烧的树烧掉
	for(int j = 1; j <= m; j++) 
		vis[cor(i, j)] = max(vis[cor(i, j)], -1+max(vis[cor(i-1, j-1)], max(vis[cor(i-1, j)], vis[cor(i, j-1)])));

	for(int i = 1; i <= n; i++) //判断是否所有的树都被烧到
	for(int j = 1; j <= m; j++) if(a[cor(i, j)] && vis[cor(i, j)] <= 0) return 0;
	return 1;
}

int main()
{
	scanf("%d%d", &n, &m);
	for(int i = 1; i <= n; i++)
	{
		getchar();
		for(int j = 1; j <= m; j++) a[cor(i, j)] = getchar() == 'X';
	}
	for(int i = n; i; i--) //求解最远燃烧距离
	for(int j = m; j; j--)
	{
		if(a[cor(i, j)])
		{
			ma[cor(i, j)] = 1 + min(ma[cor(i+1, j+1)], min(ma[cor(i+1, j)], ma[cor(i, j+1)]));
		}
	}
	int l = 0, r = 1e6; // r可以是min(n, m)
	while(l != r)
	{
		int mid = l + ((r-l) >> 1) + 1;
		if(check(mid)) l = mid;
		else r = mid-1;
	}
	int ans = l;
	printf("%d\n", ans);
	for(int i = 1; i <= n; i++) //构造答案
	for(int j = 1; j <= m; j++) if(ma[cor(i, j)] >= ans*2+1) lis[cor(i+ans, j+ans)] = 1;
	for(int i = 1; i <= n; i++)
	{
		for(int j = 1; j <= m; j++) putchar(lis[cor(i, j)] ? 'X' : '.');
		putchar(10);
	}
	return 0;
}

B. Optimal Subsequences (Easy/Hard Version) (3s 256Mb)

题目大意

nnn个aia_iai​有序，mmm个询问，每个询问有kkk和pospospos，求将前kkk大的aia_iai​按原序排列（即subsequence）（如果相同，取排出序列字典序最小），问此序列中第pospospos个元素是多少。

n≤102(E)/2⋅105(H)n \leq 10^2(E)/2\cdot 10^5(H)n≤102(E)/2⋅105(H)
ai≤109a_i \leq 10^9ai​≤109
m≤102(E)/2⋅105(H)m \leq 10^2(E)/2\cdot 10^5(H)m≤102(E)/2⋅105(H)
1≤k≤n,1≤pos≤k1 \leq k \leq n, 1 \leq pos \leq k1≤k≤n,1≤pos≤k

分析

Easy：暴力构造所有子序列即可。

Hard：线段树。离线按kkk排序询问。将aia_iai​先按大小排序，再按下标排序（字典序，小的靠前）。处理每个kkk询问，将前kkk大aia_iai​下标加入线段树，求位置为pospospos的元素。

代码

int n, m, b[maxn], t[maxn<<2];
void add(int k, int l, int r, int pos)
{
	if(pos < l || r < pos) return;
	t[k]++;
	if(l == r) return;
	add(lson, l, mid, pos); add(rson, mid+1, r, pos); return;
}

int ask(int k, int l, int r, int y)
{
	if(l == r) return l;
	if(t[lson] >= y) return ask(lson, l, mid, y);
	else return ask(rson, mid+1, r, y-t[lson]);
}

int cmp(Query aa, Query bb) { return aa.id < bb.id; }

int main()
{
	n = read();
	for(int i = 1; i <= n; i++) b[i] = a[i].v = read(), a[i].id = i;
	sort(a+1, a+n+1);
	m = read();
	for(int i = 1; i <= m; i++) q[i].k = read(), q[i].pos = read(), q[i].id = i;
	sort(q+1, q+m+1);
	int tmp = 0;
	for(int i = 1; i <= m; i++)
	{
		while(q[i].k > tmp) add(1, 1, n, a[++tmp].id);
		q[i].ans = ask(1, 1, n, q[i].pos);
	}
	sort(q+1, q+m+1, cmp);
	for(int i = 1; i <= m; i++) printf("%d\n", b[q[i].ans]);
	return 0;
}
