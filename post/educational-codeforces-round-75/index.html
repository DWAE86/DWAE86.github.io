C. Minimize The Integer (2s 256Mb)

题目大意

给你nnn长0-9串，如果相邻digit一奇数一偶数则可以交换位置（同为奇数或同为偶数不行）。求可以得到的字典序最小串。

n≤3⋅105n \leq 3\cdot 10^5n≤3⋅105

分析

分析性质。由题意，奇数字串和偶数字串各自顺序不会变（必要性），满足这一性质的串可以任意构造（充分性），随便贪（归并）。

代码

const int maxn = 3e5+10;
int n, a[maxn], c[maxn], d[maxn]; char s[maxn];
int main()
{
	int T = read();
	while(T--)
	{
		scanf("%s", s+1);
		int n = strlen(s+1);
		for(int i = 1; i <= n; i++) a[i] = s[i]-'0';
		int tot1 = 0, tot2 = 0;
		for(int i = 1; i <= n; i++) 
		{
			if(a[i]&1) c[++tot1] = a[i];
			else d[++tot2] = a[i];
		}
		int x = 1, y = 1;
		while(x <= tot1 && y <= tot2) 
		{
			if(c[x] < d[y]) putchar(c[x++]+'0');
			else putchar(d[y++]+'0');
		}
		while(x <= tot1) putchar(c[x++]+'0');
		while(y <= tot2) putchar(d[y++]+'0');
		putchar(10);
	}
	return 0;
}

D. Salary Changing (3s 256Mb)

题目大意

给nnn个人发工资，一共有sss块钱。每个人给的钱必须在[li,ri]\left[ l_i, r_i \right][li​,ri​]之间。使给的nnn个钱数中位数最大，求中位数最大可以是多少。

n≤2⋅105n \leq 2\cdot 10^5n≤2⋅105, nnn is odd
s≤2⋅1014s \leq 2\cdot 10^{14}s≤2⋅1014
li≤ri≤109l_i \leq r_i \leq 10^9li​≤ri​≤109

分析

一开始考虑了二分但是觉得不满足全域的单调性。实际上考虑了L和R的初值以后就满足。

check部分，对于每个中位数，分三种人。

  1. ri<midr_i < midri​<mid，这种人只能作为中位数前面，给他最少的l_i即可。
  2. mid≤limid \leq l_imid≤li​，这种人只能作为中位数后面，给他最少的l_i 。
  3. li<mid≤ril_i < mid \leq r_ili​<mid≤ri​，这种人既可以做前面也可以做后面，每个人要么给lil_ili​要么给midmidmid，根据lil_ili​排序即可。

最后钱不够用或者前两种人每种太多都不行。

代码

const int maxn = 2e5+10;
pair<ll, ll> a[maxn];
ll s;
int n, vis[maxn];

int check(long long mid)
{
	ll tmp = 0;
	int c1 = 0, c2 = 0;
	for(int i = 1; i <= n; i++) vis[i] = 0;
	for(int i = 1; i <= n; i++) if(a[i].second < mid) 
	{
		vis[i] = 1; tmp += a[i].first; c1++;
	}
	//if(c1 > n/2) return 0;
	for(int i = 1; i <= n; i++) if(!vis[i] && a[i].first >= mid)
	{
		vis[i] = 1; tmp += a[i].first; c2++;
	}
	for(int i = 1; i <= n && c1 != n/2; i++) if(!vis[i])
	{
		vis[i] = 1; tmp += a[i].first; c1++;
	}
	for(int i = 1; i <= n; i++) if(!vis[i])
	{
		tmp += mid; c2++;
	}
	return tmp <= s && c1 == n/2 && c2 == n/2+1;
}

int main()
{
	int T = read();
	while(T--)
	{
		n = read(); s = read();
		for(int i = 1, u, v; i <= n; i++)
		{
			u = read(), v = read();
			a[i] = {u, v};
		}
		sort(a+1, a+n+1);
		ll l = a[n/2+1].first, r = s;
		while(l != r)
		{
			ll mid = l + ((r-l)>>1) + 1;
			if(check(mid)) l = mid;
			else r = mid-1;
		}
		printf("%I64d\n", l);
	}
	return 0;
}

E. Voting (Easy/Hard Version) (2s 256Mb)

题目大意

你需要让nnn个人全部给你投票。对于每个人iii，要么花pip_ipi​的钱让他为你投票，要么如果已经有个mim_imi​人为你投票了，他会免费为你投票，也就是说投票的过程是有渐进的。求让全部为你投票的最小花费。

n≤2⋅105n \leq 2\cdot 10^5n≤2⋅105

分析

将mim_imi​相等的人归为一类，并按照mim_imi​值从大到小考虑（降序很重要）。

简单证明：对于最大mim_imi​，假如将剩下的比mim_imi​小的都买了也不够最大mim_imi​的话，必须从这最大mim_imi​里买一些人，否则可以缩小问题范围为除去是最大mim_imi​的人的问题。既然要买不如先买，然后也可以缩小问题范围。之后第二大mim_imi​，第三大mim_imi​。。。

注意之前的缩小后去除的人也可以买，计算够不够当前mim_imi​时也要将这些提前买的mim_imi​加上。

被题解带着用了multiset，其实也可以用优先队列的。

代码

int main()
{
	int T = read();
	while(T--)
	{
		n = read();
		for(int i = 1; i <= n; i++) 
		{
			a[i].m = read(), a[i].p = read();
		}
		sort(a+1, a+n+1, cmp);
		int l = 1, r, cnt = 0;
		long long ans = 0;
		while(l <= n)
		{
			r = l; while(r < n && a[r+1].m == a[l].m) r++;
			for(int i = l; i <= r; i++) s.insert(a[i]);
			if(n-r+cnt < a[l].m)
			{
				int cha = a[l].m - (n-r+cnt);
				cnt += cha;
				while(cha--)
				{
					ans += (long long) (*s.begin()).p;
					s.erase(s.begin());
				}
			}
			l = r+1;
		}
		printf("%I64d\n", ans);
		if(T) s.erase(s.begin(), s.end());
	}
	return 0;
}
