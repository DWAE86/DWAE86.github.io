A. Rikka with Minimum Spanning Trees （思维）

题目大意

随机生成一幅图，求最小生成树权和与最小生成树个数乘积。

n≤105n \leq 10^5n≤105
m=105m = 10^5m=105
1≤u,v≤n,0≤w≤264−11 \leq u,v \leq n, 0 \leq w \leq 2^{64}-11≤u,v≤n,0≤w≤264−1(unsigned long long)

分析

数据随机生成，有一棵以上最小生成树概率极小，故直接求解最小生成树。

G. Rikka with Intersections of Paths（LCA+树上差分+组合计数）

题目大意

nnn 个点的树，给定 mmm 条树上路径。从中选取 kkk 条路径，问有多少种选取方法能使得这 kkk 条路径有公共点。答案对 109+710^9+7109+7 取模。

1≤n,m≤2⋅1061 \leq n,m \leq 2\cdot10^61≤n,m≤2⋅106
2≤k≤m2 \leq k \leq m2≤k≤m

样例输入输出

1
3 6 2
1 2
1 3
1 1
2 2
3 3
1 2
1 3
2 3

10

分析

对每个点求贡献。如果某个点是 kkk 条路径的公共点，我们可以预处理出经过每个点的路径数。设有 SSS 条路径经过该点。我们接着考虑去重问题。对于任意条树上路径，如果它们有交点，那么一定有且只有一个交点是某条路径的LCA。因此我们可以只考虑这个点作为LCA的贡献，设该点是 ppp 条路径的LCA：

(Sk)−(S−pk)\begin{pmatrix} S\cr k \end{pmatrix}- \begin{pmatrix} S-p\cr k \end{pmatrix} (Sk​)−(S−pk​)

我们可以用树上差分得到经过每个点的路径数（两个端点+1，LCALCALCA 和 fa[LCA]fa[LCA]fa[LCA] -1 ，然后对该点dfs）。这一过程中处理好LCA个数即可。

代码

ll T, n, m, k;
ll num[N], cnt[N], tim[N];
// 树上差分数组，经过这个点的路径数，LCA是这个点的路径数

int head[N], tot;
struct Edge {
    int nex, to;
}e[N << 1];
void add(int a, int b) {
    e[++tot] = (Edge) {head[a], b};
    head[a] = tot;
}

int lg[N]; // log2(i)+1
ll ifac[N], fac[N];
void init_all() {
    for (int i = 1; i < N; i++) {
        lg[i] = lg[i-1] + ((1 << lg[i-1]) == i);
    }
    ifac[0] = ifac[1] = 1; fac[0] = fac[1] = 1;
    for(int i = 2; i < N; i++) {
        ifac[i] = (P - P / i) * ifac[P % i] % P;
    }
    for(int i = 2; i < N; i++) {
        ifac[i] = ifac[i] * ifac[i-1] % P;
    }
    for(int i = 2; i < N; i++) {
        fac[i] = fac[i-1] * i % P;
    }
}

ll C(ll n, ll m)
{
    if(n < m) return 0;
    return fac[n] * ifac[n - m] % P * ifac[m] % P;
}

int fa[N][22], dep[N];
void dfs(int u, int F)
{
    dep[u] = dep[F] + 1;
    fa[u][0] = F;
    for(int i = 1; (1 << i) <= dep[u]; i++) {
        fa[u][i] = fa[fa[u][i-1]][i-1];
    }
    for(int i = head[u]; i; i = e[i].nex) {
        int to = e[i].to;
        if(to != F) dfs(to, u);
    }
}
int lca(int x, int y)
{
    if(dep[x] < dep[y]) swap(x, y);
    while(dep[x] > dep[y]) x = fa[x][lg[dep[x] - dep[y]] - 1];
    if(x == y) return x;
    for(int i = lg[dep[x]] - 1; i >= 0; i--)
    {
        if(fa[x][i] != fa[y][i])
        {
            x = fa[x][i];
            y = fa[y][i];
        }
    }
    return fa[x][0];
}

void dfs2(int u, int F) {
    cnt[u] = num[u];
    for (int i = head[u]; i; i = e[i].nex) {
        int to = e[i].to;
        if (to == F) {
            continue;
        }
        dfs2(to, u);
        cnt[u] += cnt[to];
    }
}

void clear() {
    for (int i = 0; i <= n; i++) {
        head[i] = cnt[i] = num[i] = tim[i] = dep[i] = 0;
    }
    memset(fa, 0, sizeof(fa));
    tot = 0;
}

int main(int argc, const char * argv[]) {

    ios::sync_with_stdio(false);
    cin.tie(0);

    cin >> T;
    init_all();
    while (T--) {
        ll ans = 0;
        cin >> n >> m >> k;
        for (int i = 1, a, b; i < n; i++) {
            cin >> a >> b;
            add(a, b); add(b, a);
        }
        dfs(1, 0);
        for (int i = 1, a, b; i <= m; i++) {
            cin >> a >> b;
            num[a]++; num[b]++;
            int c = lca(a, b), d = fa[c][0];
            num[c]--; num[d]--;
            tim[c]++;
        }
        dfs2(1, 0);
        for (int i = 1; i <= n; i++) {
            (ans += C(cnt[i], k) - C(cnt[i] - tim[i], k) + P) %= P;
        }
        cout << ans << "\n";
        clear();
    }
    return 0;
}

H. Rikka with A Long Colour Palette（贪心+思维）

题目大意

有kkk种颜色，同时喷在一个单位块上可以生成一个DREAM。在一个000到10910^9109的数轴上，给出nnn个segment的两点坐标l,rl,rl,r，每个区域可以喷一种颜色。求DREAM个数最多是多少，输出任意一种喷颜色的方案。

n≤2⋅106n \leq 2\cdot10^6n≤2⋅106
k≤2⋅105k \leq 2\cdot10^5k≤2⋅105
0≤l,r≤1090 \leq l,r \leq 10^90≤l,r≤109

分析

答案上界为被kkk个及以上线段覆盖的单位块的个数。以下算法保证上界满足。
从左到右扫描：

  • 入点
    1. 如过当前覆盖线段小于kkk，则任意选择一种颜色。
    2. 如果当前覆盖线段已经有kkk，则先不分配颜色而加入等待。
  • 出点
    1. 如果出点线段是正在等待，则随便分配颜色。
    2. 如果出点线段已有颜色，选择任意一个等待的线段，将其颜色分配为出点所在线段的颜色。

代码

struct Node
{
	int f, x, id;
	bool operator < (const Node &u)
	{
		return x < u.x || (x == u.x && f > u.f);
	}
} a[maxn];

int n, m;
int ans, vis[maxn];
int bc[maxn], cnt; 
int an[maxn];

queue<int> q; int in[maxn];
int ask()
{
	while(!q.empty() && in[q.front()] != 2) q.pop();
	return q.empty() ? -1 : q.front();
}

int main()
{
	int T = read(); while(T--)
	{
		n = read(); m = read();
		for(int i = 1, l, r; i <= n; i++) 
		{
			l = read(), r = read();
			a[i*2-1] = (Node) {1, l, i};
			a[i*2] = (Node) {2, r, i};
		}
		n *= 2;
		sort(a+1, a+n+1);
		
		ans = cnt = 0;
		for(int i = 1; i <= m; i++) bc[++cnt] = m-i+1;
		for(int i = 1; i <= n; i++)
		{
			if(!cnt) ans += a[i].x-a[i-1].x; // got all the color 
			if(a[i].f == 1)
			{
				if(cnt) // has unpainted color
				{
					int col = bc[cnt--];
					an[a[i].id] = col;
					in[a[i].id] = 1;
				}
				else
				{
					q.push(a[i].id);
					in[a[i].id] = 2; // waiting to be painted later
				}
			}
			else
			{
				if(in[a[i].id] == 2) // paint it any color
				{
					an[a[i].id] = 1;
					in[a[i].id] = 0;
				} 
				else 
				{
					int col = an[a[i].id], j = ask(); // any segment is waiting
					if(j == -1) bc[++cnt] = col;
					else 
					{
						in[j] = 1;
						an[j] = col;
					}
				}
			}
		}
		printf("%d\n%d", ans, an[1]);
		n /= 2;
		for(int i = 2; i <= n; i++) printf(" %d", an[i]);
		putchar(10);
	}
	return 0;
}

I. Rikka with Sorting Networks（搜索）

题目大意

定义操作为一对数u，vu，vu，v，使得序列aaa中第uuu个元素aua_uau​和第vvv个元素ava_vav​按照不降序排列。定义操作序列为一系列kkk个操作，按顺序依次执行。求在长度为nnn的所有排列中，经过操作序列后，变成“almost sorted”排列。“almost sorted”排列指最长上升子序列至少为(n−1)(n-1)(n−1)的nnn排列。

1≤n≤501 \leq n \leq 501≤n≤50
0≤k≤100 \leq k \leq 100≤k≤10
1≤u<v≤n1 \leq u < v \leq n1≤u<v≤n

分析

最终“almost sorted”排列一共有O(n2)O(n^2)O(n2)（(n−1)∗(n−1)+1(n-1)*(n-1)+1(n−1)∗(n−1)+1）种。找出所有可能最终状态，按照排序序列倒着搜索。
1. 如果aua_uau​小于ava_vav​，则不可能经过这一操作后成为当前排列，退出。
2. 如果aua_uau​不小于ava_vav​，分别对经过此操作前，aua_uau​和ava_vav​无序和已经有序两种情况进行搜索。

某一开始序列操作到最后只有一种可能最终排列，故不同最终排列不可能倒着搜索到相同排列。同样，某一开始序列在每个操作是否反转只有一种可能，故同一最终排列搜索时也不能得到相同排列。故答案无需去重，只要搜到一种开始序列ans++。

！！！搜索时要么使用全局变量要么传址，否则每次dfs复制，复杂度多一个n。

代码

int n, k, mod, ans;
vector<string> st;
vector<pair<int, int> > op;

void pre()
{
	ans = 0; st.resize(0); op.resize(0);
	for(int x = 1; x <= n; x++) // place x in other position
	{
		string a(1, '0'+x), b(1, '0'+x);
		for(int i = 1; i <= n; i++) if(i != x) b += '0'+i;
		for(int i = 1; i <= n; i++) if(i != x-1 && i != x) // not before x-1 or x+1
		{
			st.push_back(b.substr(1, i-1)+a+b.substr(i, n-i));
		}
		if(x == 1) st.push_back(a+b.substr(1, n-1));
	}
//	for(string s : st) cout << s+"\n";
}

void dfs(int x, string &s)
{
	if(x == k) 
	{
		ans = (ans+1)%mod;
		return;
	}
	int u = op[x].first, v = op[x].second;
	if(s[u-1] > s[v-1]) return;
	dfs(x+1, s);
	swap(s[u-1], s[v-1]); 
	dfs(x+1, s);
	swap(s[u-1], s[v-1]); 
}

int main()
{
	int T = read(); while(T--)
	{
		n = read(), k = read(), mod = read();
		pre();
		for(int i = 1; i <= k; i++) op.push_back({read(), read()});
		reverse(op.begin(), op.end());
		for(string &s : st) dfs(0, s);
		printf("%d\n", ans);
	}
	return 0;
}
